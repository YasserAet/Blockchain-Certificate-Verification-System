\section{Vue d'Ensemble du Système}
% ============================================================

\begin{frame}{Objectifs du Projet}
  \textbf{Objectif}
  
  \vspace{0.3cm}
  
  Construire un système pour vérifier les certificats académiques et professionnels en utilisant la technologie blockchain et l'intelligence artificielle.
  
  \vspace{0.5cm}
  
  \textbf{Fonctionnalités Clés}
  \begin{itemize}
    \item Stockage des certificats sur blockchain
    \item Vérification de documents assistée par IA
    \item Détection de fraude par apprentissage automatique
    \item Extraction de compétences des certificats
    \item Vérification décentralisée
  \end{itemize}
\end{frame}

\begin{frame}{Architecture du Système}
  \textbf{Composants Principaux}
  
  \vspace{0.5cm}
  
  \begin{itemize}
    \item \textbf{API Backend} - Node.js avec TypeScript
    \item \textbf{Smart Contracts} - Blockchain Ethereum
    \item \textbf{Service ML} - Python avec modèles pré-entraînés
    \item \textbf{Frontend} - Interface web basée sur React
    \item \textbf{Base de Données} - PostgreSQL pour les métadonnées
  \end{itemize}
\end{frame}

\begin{frame}{Smart Contracts Implémentés}
  \textbf{Trois Contrats Principaux}
  
  \vspace{0.5cm}
  
  \begin{enumerate}
    \item \textbf{CertificateRegistry}
    \begin{itemize}
      \item Émet des certificats sur la blockchain
      \item Vérifie l'authenticité des certificats
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{FraudDetectionStore}
    \begin{itemize}
      \item Enregistre les résultats de détection de fraude
      \item Crée une piste d'audit immuable
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{SkillValidator}
    \begin{itemize}
      \item Valide les compétences des certificats
      \item Lie les compétences aux enregistrements blockchain
    \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}{Implémentation de l'Apprentissage Automatique}
  \textbf{Trois Services IA Développés}
  
  \vspace{0.5cm}
  
  \begin{enumerate}
    \item \textbf{Classification de Documents}
    \begin{itemize}
      \item Identifie si un document est un certificat
      \item Utilise le modèle pré-entraîné ResNet
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{Extraction de Texte OCR}
    \begin{itemize}
      \item Extrait le texte des images de certificats
      \item Support multi-langues
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{Extraction de Compétences}
    \begin{itemize}
      \item Identifie les compétences mentionnées dans les certificats
      \item Utilise la correspondance de similarité sémantique
    \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}{Modèles IA Expliqués}
  \textbf{Modèles Pré-entraînés Utilisés}
  
  \vspace{0.5cm}
  
  \begin{enumerate}
    \item \textbf{ResNet (Residual Network)}
    \begin{itemize}
      \item Modèle d'apprentissage profond pour l'analyse d'images
      \item Détecte la fraude visuelle dans les certificats
      \item Analyse les motifs, logos et signatures
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{LayoutLM (Layout Language Model)}
    \begin{itemize}
      \item Spécialisé pour la compréhension de documents
      \item Extrait des données structurées des certificats
      \item Combine texte, mise en page et informations visuelles
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{XLM-RoBERTa (Multilingual Model)}
    \begin{itemize}
      \item Compréhension linguistique pour plus de 100 langues
      \item Extrait les compétences du texte des certificats
      \item Fonctionne avec les certificats en anglais, français, arabe
    \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}{Datasets Identifiés}
   \textbf{Trois Datasets Principaux Sélectionnés}
  
  \vspace{0.5cm}
  
  \begin{enumerate}
    \item \textbf{Certificate Dataset (Kaggle)}
    \begin{itemize}
      \item 1000+ images de certificats
      \item Pour l'entraînement de la classification de documents
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{Coursera Courses Dataset}
    \begin{itemize}
      \item 50,000+ cours avec informations sur les compétences
      \item Pour l'extraction et la cartographie des compétences
    \end{itemize}
    
    \vspace{0.3cm}
    
    \item \textbf{CASIA Image Tampering Dataset}
    \begin{itemize}
      \item 12,000+ images authentiques et altérées
      \item Pour l'entraînement de la détection de fraude
    \end{itemize}
  \end{enumerate}
\end{frame}
